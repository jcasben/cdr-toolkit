/// Takes the user input and parses it to a Result<Vec<f32>, &str>.
///
/// # Arguments
/// * `message` - the message that will be showed to the user.
///
/// # Returns
/// * `Ok(Vec<f32>)` - if there was no error at the parsing.
/// The Ok contains the resulting Vec<f32> parsed from the user
/// input.
/// * `Err(&str)` - if some error occurred during the parsing.
/// The Err contains the message of the error.
pub fn parse_user_input(message: &str) -> Result<Vec<f32>, &str> {
    let mut user_input = String::new();
    print!("\n{}",message.blue());
    io::stdout().flush().unwrap();
    io::stdin().read_line(&mut user_input).expect("ERROR: couldn't read the user input!");

    let result: Result<Vec<f32>, &'static str> = user_input
        .split(',')
        .map(|s| s.trim().parse::<f32>().map_err(|_| "Couldn't parse to f32"))
        .collect();

    result.and_then(|v| {
        if v.iter().any(|&x| x.is_nan()) {
            Err("Found an element that couldn't be parsed")
        } else if v.is_empty() {
            Err("No valid elements found")
        } else {
            Ok(v)
        }
    })
}
